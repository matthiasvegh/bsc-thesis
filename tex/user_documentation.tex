\newread\ownread
\newcommand{\piprequirements}[1]{
	\IfFileExists{#1}{
		\openin\ownread=#1
		\begin{itemize}
			\loop\unless\ifeof\ownread
			\read\ownread to \requirement
			\ifeof\ownread
			\else
				\item \requirement \cite{\requirement}
			\fi
			\repeat
		\end{itemize}
		\closein\ownread
	}{File not found}
}
